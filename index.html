/** * GOOGLE APPS SCRIPT FOR QUIZ RESULTS * * SETUP INSTRUCTIONS: * 1. Open your Google Sheet: https://docs.google.com/spreadsheets/d/1jGa3gP4rCwZR_NcpPmmFrlXZok1zuPsjeZ2OnWuevWg/edit * 2. Click on "Extensions" > "Apps Script" * 3. Delete any existing code * 4. Copy and paste this entire code * 5. Click the disk icon to save * 6. Click "Deploy" > "New deployment" * 7. Click the gear icon next to "Select type" and choose "Web app" * 8. Set the following: * - Description: Quiz Results Collector * - Execute as: Me * - Who has access: Anyone * 9. Click "Deploy" * 10. Copy the Web App URL and send it back to me * 11. Click "Authorize access" and grant permissions */ function doPost(e) { try { // Parse the incoming data const data = JSON.parse(e.postData.contents); // Get the active spreadsheet const ss = SpreadsheetApp.getActiveSpreadsheet(); // Create or get the Summary sheet let summarySheet = ss.getSheetByName('Summary'); if (!summarySheet) { summarySheet = ss.insertSheet('Summary'); // Add headers summarySheet.appendRow([ 'Timestamp', 'Student Name', 'Score', 'Total Questions', 'Percentage', 'Grade' ]); // Format headers summarySheet.getRange('A1:F1').setFontWeight('bold').setBackground('#4a90e2').setFontColor('#ffffff'); } // Calculate grade const percentage = data.percentage; let grade = 'F'; if (percentage >= 90) grade = 'A'; else if (percentage >= 80) grade = 'B'; else if (percentage >= 70) grade = 'C'; else if (percentage >= 60) grade = 'D'; // Add summary data summarySheet.appendRow([ data.timestamp, data.studentName, data.score, data.totalQuestions, data.percentage + '%', grade ]); // Create or get detailed results sheet for this student const sheetName = data.studentName + ' - ' + new Date().toLocaleDateString(); let detailSheet = ss.getSheetByName(sheetName); if (!detailSheet) { detailSheet = ss.insertSheet(sheetName); // Add headers detailSheet.appendRow([ 'Question #', 'Section', 'Question', 'Correct Answer', 'Student Answer', 'Result' ]); // Format headers detailSheet.getRange('A1:F1').setFontWeight('bold').setBackground('#4a90e2').setFontColor('#ffffff'); } // Add detailed results data.results.forEach(function(result) { detailSheet.appendRow([ result.questionNumber, result.section, result.question, result.correctAnswer, result.studentAnswer, result.isCorrect ? '✓ Correct' : '✗ Incorrect' ]); // Color code the result const lastRow = detailSheet.getLastRow(); if (result.isCorrect) { detailSheet.getRange(lastRow, 6).setBackground('#d5f4e6').setFontColor('#27ae60'); } else { detailSheet.getRange(lastRow, 6).setBackground('#fadbd8').setFontColor('#e74c3c'); } }); // Auto-resize columns detailSheet.autoResizeColumns(1, 6); summarySheet.autoResizeColumns(1, 6); return ContentService.createTextOutput(JSON.stringify({ 'result': 'success', 'message': 'Data saved successfully' })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ 'result': 'error', 'message': error.toString() })).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput('Quiz Results API is running. Use POST to submit data.'); }
